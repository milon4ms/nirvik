// 🚪 সরাসরি ও সহজ লগআউট অ্যাকশন (অটো-লগইন প্রতিরোধসহ) window.addEventListener('click', function(e) { if (e.target && e.target.closest('.nh-logout')) { e.preventDefault(); if (confirm('আপনি কি নিশ্চিতভাবে লগআউট করতে চান?')) { try { if (typeof firebase !== 'undefined' && firebase.auth) { // ১. ফায়ারবেস থেকে সাইন আউট firebase.auth().signOut().then(function() { // ২. ব্রাউজারের লোকাল মেমোরি ও সেশন ডেটা সম্পূর্ণ ক্লিয়ার করা if (typeof localStorage !== 'undefined') localStorage.clear(); if (typeof sessionStorage !== 'undefined') sessionStorage.clear(); alert('✅ সফলভাবে লগআউট হয়েছে!'); // ৩. ক্যাশ এড়াতে সম্পূর্ণ ফ্রেশভাবে লগইন পেজে পাঠানো window.location.replace('login.html'); }).catch(function(error) { alert('❌ লগআউট করতে সমস্যা হয়েছে: ' + error.message); }); } else { window.location.replace('login.html'); } } catch (err) { window.location.replace('login.html'); } } } });
