// ai-prompt-sender.js - কেন্দ্রীয় স্ক্রিপ্ট (যেকোনো পেজে ব্যবহারযোগ্য) (function() { 'use strict'; // কনফিগারেশন সেটিংস const CONFIG = { aiPlatform: 'perplexity', // perplexity, chatgpt, gemini, grok autoSubmit: true, debugMode: true }; // AI প্ল্যাটফর্মের URL টেমপ্লেট const AI_URLS = { perplexity: (prompt) => `https://www.perplexity.ai/?q=${encodeURIComponent(prompt)}`, chatgpt: (prompt) => `https://chat.openai.com/?q=${encodeURIComponent(prompt)}`, gemini: (prompt) => `https://gemini.google.com/?q=${encodeURIComponent(prompt)}`, grok: (prompt) => `https://grok.com/?q=${encodeURIComponent(prompt)}` }; // লগিং ফাংশন (ডিবাগিং এর জন্য) function log(message, type = 'info') { if (!CONFIG.debugMode) return; const prefix = '🤖 [AI-Prompt-Sender]'; if (type === 'error') console.error(prefix, message); else if (type === 'success') console.log(prefix, message); else console.log(prefix, message); } // ফর্ম ডাটা কালেক্ট করার ফাংশন function collectFormData() { let formData = []; // সব সেকশন হেডার সংগ্রহ const sections = document.querySelectorAll('.section'); sections.forEach(section => { const header = section.querySelector('h3'); const sectionTitle = header ? header.innerText.trim() : 'অন্যান্য তথ্য'; // টেক্সটএরিয়া ও ইনপুট ফিল্ডের মান নেওয়া const textareas = section.querySelectorAll('textarea, input[type="text"]'); textareas.forEach(input => { if (input.value && input.value.trim() !== '') { formData.push(`${sectionTitle}: ${input.value.trim()}`); } }); // চেকবক্সের মান নেওয়া const checkboxes = section.querySelectorAll('input[type="checkbox"]:checked'); if (checkboxes.length > 0) { const checkedValues = Array.from(checkboxes).map(cb => cb.value).join(', '); formData.push(`${sectionTitle}: ${checkedValues}`); } }); return formData; } // প্রম্পট জেনারেট করার ফাংশন function generatePrompt() { const formData = collectFormData(); if (formData.length === 0) { log('কোনো ডাটা পাওয়া যায়নি!', 'error'); return null; } const prompt = `আমি নিচে আমার একজন রোগীর রোগের লক্ষণসমূহ লিখছি। তুমি লক্ষণসমূহ হোমিওপ্যাথিক রেপার্টরি গুলিতে যাচাই করে ঔষধ সিলেক্ট করে দাও। লক্ষণসমূহ: ${formData.join('\n')} নির্দেশনা: ১) প্রধান হোমিওপ্যাথিক ঔষধের নাম (যেটি সবচেয়ে বেশি উপযুক্ত) ২) প্রধান ঔষধের সহায়ক ঔষধের নাম (Complementary) ৩) সবচেয়ে বেশি উপযুক্ত বায়োকেমিক ও বায়োকম্বিনেশনের নাম ৪) লক্ষণ সমুহের সাথে মিল আছে এমন ৬/৭ ঔষধের তালিকা ৫) এলোপ্যাথি ও ইউনানি তে এই রোগের ঔষধ (সংক্ষেপে) দয়া করে এমন ঔষধ নির্বাচন করবেন না যারা একে অপরের ক্রিয়া নাশক বা শত্রুভাবাপন্ন।`; log('প্রম্পট জেনারেট সম্পন্ন', 'success'); return prompt; } // AI তে পাঠানোর ফাংশন function sendToAI() { log('প্রম্পট জেনারেট করা শুরু...'); const prompt = generatePrompt(); if (!prompt) { alert('❌ কোনো ডাটা পাওয়া যায়নি! দয়া করে ফর্ম পূরণ করুন।'); return; } // প্রম্পট প্রিভিউ দেখান (যদি result div থাকে) const resultDiv = document.getElementById('result'); if (resultDiv) { resultDiv.innerHTML = `📝 জেনারেটেড প্রম্পট:

${prompt.replace(/\n/g, '
')}`; } // AI প্ল্যাটফর্মে পাঠানো const urlGenerator = AI_URLS[CONFIG.aiPlatform]; if (urlGenerator) { const aiUrl = urlGenerator(prompt); log(`খোলা হচ্ছে: ${aiUrl}`); window.open(aiUrl, '_blank'); } else { log('অজানা AI প্ল্যাটফর্ম', 'error'); } } // বিদ্যমান সাবমিট বাটন খুঁজে তার সাথে যুক্ত করা function integrateWithExistingButton() { // ফর্মের সাবমিট বাটন খোঁজা const submitBtn = document.querySelector('button[type="submit"]'); const copyBtn = document.getElementById('copyButton'); if (submitBtn) { // মূল সাবমিট ফাংশন সংরক্ষণ const originalClick = submitBtn.onclick; // নতুন ফাংশন যোগ করা submitBtn.addEventListener('click', function(e) { // প্রথমে ফর্ম ডাটা প্রদর্শন হবে (result div এ) // তারপর AI তে পাঠানো হবে setTimeout(() => { sendToAI(); }, 100); // সামান্য delay যেন result আগে আপডেট হয় }); log('সাবমিট বাটনের সাথে যুক্ত করা হয়েছে', 'success'); } else { log('সাবমিট বাটন পাওয়া যায়নি!', 'error'); } } // কাস্টম বাটন তৈরি করা (যদি দরকার হয়) function addCustomButton() { // চেক করা ইতিমধ্যে বাটন আছে কিনা if (document.getElementById('aiPromptBtn')) return; const buttonContainer = document.querySelector('.submit-area') || document.querySelector('form').querySelector('button')?.parentNode || document.querySelector('form'); if (buttonContainer) { const aiBtn = document.createElement('button'); aiBtn.id = 'aiPromptBtn'; aiBtn.innerHTML = '🤖 AI তে প্রম্পট পাঠান'; aiBtn.style.cssText = ` background: linear-gradient(135deg, #667eea 0%, #764ba2 100%); color: white; border: none; padding: 12px 24px; margin-left: 10px; border-radius: 6px; cursor: pointer; font-size: 16px; font-weight: bold; `; aiBtn.onclick = (e) => { e.preventDefault(); sendToAI(); }; buttonContainer.appendChild(aiBtn); log('কাস্টম বাটন যোগ করা হয়েছে', 'success'); } } // পেজ লোড হলে সবকিছু সক্রিয় করা function init() { log('AI প্রম্পট সেন্ডার সক্রিয় হচ্ছে...'); // ফর্ম লোড হওয়ার জন্য সামান্য অপেক্ষা setTimeout(() => { integrateWithExistingButton(); addCustomButton(); }, 500); } // পেজ লোড হলে রান করা if (document.readyState === 'loading') { document.addEventListener('DOMContentLoaded', init); } else { init(); } })();
